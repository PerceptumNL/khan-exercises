\qquad = \displaystyle \lim_{H \to 0} \frac{(expr( ["+", ["*", A, ["^", ["+", "x", H], 2]], ["*", B, ["+", "x", H]], C] )) - (expr( ["+", ["*", A, ["^", "x", 2]], ["*", B, "x"], C] ))}{H}

\qquad = \displaystyle \lim_{H \to 0} \frac{(expr( ["+", ["*", A, ["+", ["^", "x", 2], "2x " + H, ["^", H, 2]]], ["*", B, ["+", "x", H]], C] )) - (expr( ["+", ["*", A, ["^", "x", 2]], ["*", B, "x"], C] ))}{H}
